```python
# plugins/sdlc-knowledge-base/scripts/kb_config.py
"""Parse [Knowledge Base] section from CLAUDE.md; expose layer vocabulary helpers."""
from __future__ import annotations

import re
import sys
from pathlib import Path

import yaml

DEFAULT_LAYERS: list[str] = ["methodology", "evidence", "domain", "development"]

_LAYER_VALUE_RE = re.compile(r"^[a-z][a-z0-9-]*$")
_KB_HEADING_RE = re.compile(r"^##\s+Knowledge Base\s*$", re.IGNORECASE)
_ANY_HEADING_RE = re.compile(r"^##\s+")
_EXCLUDED_NAMES = frozenset({"_shelf-index.md", "_index.md", "log.md"})
_EXCLUDED_DIRS = frozenset({"raw"})


def _parse_kb_section(project_dir: Path) -> dict:
    """Return parsed key-value pairs from the ## Knowledge Base section of CLAUDE.md."""
    claude_md = project_dir / "CLAUDE.md"
    if not claude_md.exists():
        return {}

    lines = claude_md.read_text(encoding="utf-8").splitlines()
    in_section = False
    result: dict = {}
    i = 0
    while i < len(lines):
        line = lines[i]
        if _KB_HEADING_RE.match(line):
            in_section = True
            i += 1
            continue
        if in_section and _ANY_HEADING_RE.match(line):
            break
        if in_section:
            m = re.match(r"^([a-zA-Z_][\w_-]*):\s*(.*)", line)
            if m:
                key = m.group(1)
                val = m.group(2).strip()
                if val:
                    result[key] = val
                else:
                    items: list[str] = []
                    i += 1
                    while i < len(lines) and re.match(r"^\s+-\s+", lines[i]):
                        item = re.sub(r"^\s+-\s+", "", lines[i])
                        item = re.sub(r"\s*#.*$", "", item).strip()
                        if item:
                            items.append(item)
                        i += 1
                    if items:
                        result[key] = items
                    continue
        i += 1
    return result


def allowed_layers(project_dir: Path) -> list[str]:
    """Return the allowed layer set for the project.

    Returns the four defaults when CLAUDE.md has no `layers:` declaration.
    Returns the project-declared list when `layers:` is present (replaces defaults).
    """
    section = _parse_kb_section(project_dir)
    if "layers" in section and isinstance(section["layers"], list):
        return list(section["layers"])
    return DEFAULT_LAYERS[:]


def _is_library_file(path: Path, library_path: Path) -> bool:
    """Return True if path should be checked for layer compliance."""
    if path.name in _EXCLUDED_NAMES:
        return False
    try:
        rel = path.relative_to(library_path)
    except ValueError:
        return False
    if rel.parts[0] in _EXCLUDED_DIRS:
        return False
    return path.suffix == ".md"


def _read_frontmatter(text: str) -> dict:
    """Parse YAML frontmatter; return {} on failure."""
    m = re.match(r"^---\s*\n(.*?)\n---\s*\n", text, re.DOTALL)
    if not m:
        return {}
    try:
        result = yaml.safe_load(m.group(1))
        return result if isinstance(result, dict) else {}
    except yaml.YAMLError:
        return {}


def check_layer_compliance(
    library_path: Path, allowed: list[str]
) -> list[tuple[str, str]]:
    """Return list of (relative_path, error_message) for non-compliant files.

    Checks every .md library file (excluding _shelf-index.md, log.md, raw/).
    A file is non-compliant if its `layer:` frontmatter field is missing
    or holds a value not in `allowed`.
    """
    violations: list[tuple[str, str]] = []
    for md_file in sorted(library_path.rglob("*.md")):
        if not _is_library_file(md_file, library_path):
            continue
        rel = str(md_file.relative_to(library_path))
        text = md_file.read_text(encoding="utf-8")
        fm = _read_frontmatter(text)
        layer = fm.get("layer", "")
        if not layer:
            violations.append((rel, f"missing `layer:` field"))
        elif str(layer).strip().lower() not in allowed:
            violations.append((rel, f"invalid layer '{layer}' — not in allowed set {allowed}"))
    return violations


def main(args: list[str] | None = None) -> int:
    """CLI: check layer compliance for a library directory.

    Usage: kb_config.py <library_path> [--project-dir <dir>]
    Exits 0 if compliant, 1 if violations found.
    """
    import argparse

    parser = argparse.ArgumentParser(description="Check KB layer compliance.")
    parser.add_argument("library_path", help="Path to the library directory")
    parser.add_argument(
        "--project-dir",
        default=".",
        help="Project root containing CLAUDE.md (default: .)",
    )
    parsed = parser.parse_args(args)

    library_path = Path(parsed.library_path)
    project_dir = Path(parsed.project_dir)

    if not library_path.is_dir():
        print(f"Error: library path '{library_path}' does not exist.", file=sys.stderr)
        return 1

    layers = allowed_layers(project_dir)
    violations = check_layer_compliance(library_path, layers)

    if not violations:
        print(f"Layer compliance: OK ({len(layers)} allowed layers)")
        return 0

    print(f"Layer compliance: {len(violations)} violation(s)")
    for path, msg in violations:
        print(f"  {path}: {msg}")
    return 1


if __name__ == "__main__":
    sys.exit(main())
```

```python
# plugins/sdlc-knowledge-base/scripts/kb_layers.py
"""Manage the project's layer vocabulary in CLAUDE.md Knowledge Base section."""
from __future__ import annotations

import re
import sys
from pathlib import Path

from .kb_config import DEFAULT_LAYERS, _parse_kb_section, allowed_layers

_LAYER_RE = re.compile(r"^[a-z][a-z0-9-]*$")
_KB_HEADING_RE = re.compile(r"^##\s+Knowledge Base\s*$", re.IGNORECASE)
_ANY_HEADING_RE = re.compile(r"^##\s+")
_LAYER_ENTRY_RE = re.compile(r"^\*\*Layer:\*\*\s+(\S+)\s*$", re.MULTILINE)


class LayerOperationError(Exception):
    pass


def _count_layer_usage(shelf_index_path: Path) -> dict[str, int]:
    """Return count of each layer value found in the shelf-index **Layer:** entries."""
    if not shelf_index_path.exists():
        return {}
    content = shelf_index_path.read_text(encoding="utf-8")
    counts: dict[str, int] = {}
    for m in _LAYER_ENTRY_RE.finditer(content):
        val = m.group(1).strip().lower()
        counts[val] = counts.get(val, 0) + 1
    return counts


def _write_claude_md_atomic(project_dir: Path, content: str) -> None:
    claude_md = project_dir / "CLAUDE.md"
    tmp = project_dir / "CLAUDE.md.tmp"
    tmp.write_text(content, encoding="utf-8")
    tmp.rename(claude_md)


def _materialise_layers_in_claude_md(
    project_dir: Path, new_layers: list[str]
) -> None:
    """Write the `layers:` list into the ## Knowledge Base section of CLAUDE.md.

    If `layers:` already exists, replaces it. If absent, appends it after the
    last key-value line in the section. Adds the section header if missing.
    """
    claude_md = project_dir / "CLAUDE.md"
    text = claude_md.read_text(encoding="utf-8") if claude_md.exists() else ""
    lines = text.splitlines(keepends=True)

    layers_block = "layers:\n" + "".join(f"  - {layer}\n" for layer in new_layers)

    in_section = False
    section_start = -1
    section_end = len(lines)
    layers_start = -1
    layers_end = -1
    i = 0
    while i < len(lines):
        line = lines[i].rstrip("\n").rstrip("\r")
        if _KB_HEADING_RE.match(line):
            in_section = True
            section_start = i
            i += 1
            continue
        if in_section and _ANY_HEADING_RE.match(line):
            section_end = i
            break
        if in_section and re.match(r"^layers:\s*$", line):
            layers_start = i
            j = i + 1
            while j < section_end and re.match(r"^\s+-\s+", lines[j]):
                j += 1
            layers_end = j
        i += 1

    if section_start == -1:
        text = text.rstrip("\n") + "\n\n## Knowledge Base\n\n" + layers_block
        _write_claude_md_atomic(project_dir, text)
        return

    new_lines = lines[:]
    if layers_start != -1:
        new_lines[layers_start:layers_end] = [layers_block]
    else:
        new_lines.insert(section_end, layers_block)

    _write_claude_md_atomic(project_dir, "".join(new_lines))


def list_layers(project_dir: Path, shelf_index_path: Path) -> dict:
    """Return a dict with mode, allowed, usage counts, and uncategorized count."""
    section = _parse_kb_section(project_dir)
    has_explicit = "layers" in section and isinstance(section["layers"], list)
    current = section["layers"] if has_explicit else DEFAULT_LAYERS[:]

    usage = _count_layer_usage(shelf_index_path)
    uncategorized = usage.pop("uncategorized", 0)

    return {
        "mode": "project-defined" if has_explicit else "defaults",
        "allowed": current,
        "usage": {layer: usage.get(layer, 0) for layer in current},
        "uncategorized_count": uncategorized,
    }


def add_layer(project_dir: Path, layer: str) -> str:
    """Add a layer to the project's allowed set.

    Returns a message string. Raises LayerOperationError for invalid values.
    """
    if not _LAYER_RE.match(layer):
        raise LayerOperationError(
            f"invalid layer name '{layer}': must match ^[a-z][a-z0-9-]*$"
        )
    current = allowed_layers(project_dir)
    if layer in current:
        return f"no-op: '{layer}' is already in the allowed set"

    new_layers = current + [layer]
    _materialise_layers_in_claude_md(project_dir, new_layers)
    return f"added '{layer}' to the allowed layer set"


def remove_layer(
    project_dir: Path, shelf_index_path: Path, layer: str, force: bool = False
) -> str:
    """Remove a layer from the project's allowed set.

    Refuses if the layer is in use (unless force=True).
    Refuses to remove the last remaining layer.
    Raises LayerOperationError on blocked operations.
    """
    current = allowed_layers(project_dir)
    if layer not in current:
        return f"no-op: '{layer}' is not in the current allowed set"

    if len(current) == 1:
        raise LayerOperationError(
            f"cannot remove '{layer}': it is the last remaining layer. "
            "The KB requires at least one valid layer value."
        )

    if not force:
        usage = _count_layer_usage(shelf_index_path)
        count = usage.get(layer, 0)
        if count > 0:
            raise LayerOperationError(
                f"cannot remove '{layer}': it is in use by {count} file(s) "
                f"in the shelf-index. Use --force to bypass (kb-lint will "
                f"report dangling layer references)."
            )

    new_layers = [lay for lay in current if lay != layer]
    _materialise_layers_in_claude_md(project_dir, new_layers)
    return f"removed '{layer}' from the allowed layer set"


def main(args: list[str] | None = None) -> int:
    """CLI entry point for kb-layers skill."""
    import argparse

    parser = argparse.ArgumentParser(description="Manage KB layer vocabulary.")
    parser.add_argument(
        "--project-dir", default=".", help="Project root containing CLAUDE.md"
    )
    parser.add_argument(
        "--shelf-index-path", default=None, help="Path to shelf-index (for usage counts)"
    )
    group = parser.add_mutually_exclusive_group()
    group.add_argument("--add", metavar="LAYER", help="Add a layer to the allowed set")
    group.add_argument("--remove", metavar="LAYER", help="Remove a layer from the allowed set")
    parser.add_argument("--force", action="store_true", help="Bypass usage check on --remove")
    parsed = parser.parse_args(args)

    project_dir = Path(parsed.project_dir)
    shelf_path = (
        Path(parsed.shelf_index_path)
        if parsed.shelf_index_path
        else project_dir / "library" / "_shelf-index.md"
    )

    try:
        if parsed.add:
            msg = add_layer(project_dir, parsed.add)
            print(msg)
            return 0
        elif parsed.remove:
            msg = remove_layer(project_dir, shelf_path, parsed.remove, force=parsed.force)
            print(msg)
            return 0
        else:
            result = list_layers(project_dir, shelf_path)
            print(f"# Project layer set\n")
            print(f"Mode: {result['mode']}")
            print(f"\nAllowed layers:")
            for layer in result["allowed"]:
                count = result["usage"].get(layer, 0)
                print(f"  - {layer:<20} ({count} files)")
            if result["uncategorized_count"] > 0:
                print(f"\nFiles with 'uncategorized' layer: {result['uncategorized_count']}")
            return 0
    except LayerOperationError as exc:
        print(f"Error: {exc}", file=sys.stderr)
        return 1


if __name__ == "__main__":
    sys.exit(main())
```

```python
# plugins/sdlc-knowledge-base/scripts/kb_stats.py
"""Pure-Python statistics dashboard for sdlc-knowledge-base.

Reads library/_shelf-index.md and library/log.md. No LLM invocation.
"""
from __future__ import annotations

import argparse
import re
import sys
from datetime import datetime, timedelta, timezone
from pathlib import Path


_ENTRY_PATH_RE = re.compile(r"^## \d+\.\s+(.+)$", re.MULTILINE)
_LAYER_RE = re.compile(r"^\*\*Layer:\*\*\s+(\S+)\s*$", re.MULTILINE)
_TERMS_RE = re.compile(r"^\*\*Terms:\*\*\s+(.+)$", re.MULTILINE)
_FACTS_RE = re.compile(r"^\*\*Facts:\*\*\s*\n((?:-[^\n]+\n?)*)", re.MULTILINE)
_LINKS_RE = re.compile(r"^\*\*Links:\*\*\s*(.*)$", re.MULTILINE)
_LAST_REBUILT_RE = re.compile(r"^<!--\s*last_rebuilt:\s*(.+?)\s*-->", re.MULTILINE)
_LOG_ENTRY_RE = re.compile(
    r"^##\s+\[(\d{4}-\d{2}-\d{2})\]\s+(\S+)\s+\|?\s*(.*?)$", re.MULTILINE
)


def _parse_shelf_index(content: str) -> list[dict]:
    """Parse shelf-index entries; return list of dicts with file, layer, facts_count, links."""
    entries = []
    # Split on entry headings
    parts = re.split(r"^## \d+\.", content, flags=re.MULTILINE)
    for part in parts[1:]:  # first part is the header
        lines = part.strip()
        path_m = re.match(r"(.+?)$", lines, re.MULTILINE)
        file_path = path_m.group(1).strip() if path_m else "unknown"

        layer_m = _LAYER_RE.search(lines)
        layer = layer_m.group(1).strip().lower() if layer_m else "uncategorized"

        facts_m = _FACTS_RE.search(lines)
        facts_block = facts_m.group(1) if facts_m else ""
        facts_count = len([l for l in facts_block.splitlines() if l.strip().startswith("-")])
        # Don't count "(no structured findings)" as a real fact
        if "(no structured findings)" in facts_block:
            facts_count = 0

        links_m = _LINKS_RE.search(lines)
        links_str = links_m.group(1).strip() if links_m else ""
        links = [lk.strip() for lk in links_str.split(",") if lk.strip()]

        terms_m = _TERMS_RE.search(lines)
        terms_str = terms_m.group(1).strip() if terms_m else ""
        domain_candidates = [t.strip() for t in terms_str.split(",") if t.strip()]
        domain = domain_candidates[0] if domain_candidates else "unknown"

        entries.append({
            "file": file_path,
            "layer": layer,
            "domain": domain,
            "facts_count": facts_count,
            "links": links,
        })
    return entries


def _parse_log_entries(content: str) -> list[dict]:
    """Parse log.md entries; return list of dicts with date, operation, subject."""
    entries = []
    for m in _LOG_ENTRY_RE.finditer(content):
        entries.append({
            "date": m.group(1),
            "operation": m.group(2),
            "subject": m.group(3).strip(),
        })
    return entries


def _parse_last_rebuilt(content: str) -> str | None:
    m = _LAST_REBUILT_RE.search(content)
    return m.group(1).strip() if m else None


def generate_stats(
    library_path: Path,
    shelf_index_path: Path,
    log_path: Path,
    since_days: int = 30,
) -> str:
    """Generate a markdown statistics report. Read-only."""
    now = datetime.now(timezone.utc)
    now_str = now.isoformat().replace("+00:00", "Z")

    shelf_content = shelf_index_path.read_text(encoding="utf-8") if shelf_index_path.exists() else ""
    log_content = log_path.read_text(encoding="utf-8") if log_path.exists() else ""

    entries = _parse_shelf_index(shelf_content)
    log_entries = _parse_log_entries(log_content)
    last_rebuilt = _parse_last_rebuilt(shelf_content)

    total_files = len(entries)
    total_facts = sum(e["facts_count"] for e in entries)

    layer_files: dict[str, int] = {}
    layer_facts: dict[str, int] = {}
    for e in entries:
        layer = e["layer"]
        layer_files[layer] = layer_files.get(layer, 0) + 1
        layer_facts[layer] = layer_facts.get(layer, 0) + e["facts_count"]

    domain_files: dict[str, int] = {}
    for e in entries:
        domain = e["domain"]
        domain_files[domain] = domain_files.get(domain, 0) + 1

    files_lacking_layer = sum(1 for e in entries if e["layer"] == "uncategorized")
    files_lacking_xrefs = sum(1 for e in entries if not e["links"])

    cutoff = (now - timedelta(days=since_days)).strftime("%Y-%m-%d")
    recent_log = [le for le in log_entries if le["date"] >= cutoff]

    if last_rebuilt:
        try:
            rebuilt_dt = datetime.fromisoformat(last_rebuilt.replace("Z", "+00:00"))
            delta = now - rebuilt_dt
            days_ago = delta.days
            staleness_status = "✅ fresh" if days_ago <= 1 else f"⚠️ {days_ago} day(s) ago"
        except ValueError:
            staleness_status = "unknown"
            days_ago = None
    else:
        staleness_status = "unknown (no last_rebuilt header)"

    lines = [
        "# Knowledge Base Statistics",
        "",
        f"Generated: {now_str}",
        f"Library: local ({library_path}/)",
        "",
        "## Inventory",
        f"- Total files: {total_files}",
        f"- Total findings: {total_facts}",
        f"- Files lacking layer tag: {files_lacking_layer}",
        f"- Files lacking cross-references (orphans): {files_lacking_xrefs}",
        "",
        "## Distribution by layer",
        "| Layer | Files | Findings | Avg findings/file |",
        "|---|---|---|---|",
    ]
    for layer in sorted(layer_files.keys()):
        fc = layer_files[layer]
        ff = layer_facts[layer]
        avg = f"{ff/fc:.1f}" if fc > 0 else "-"
        lines.append(f"| {layer} | {fc} | {ff} | {avg} |")

    lines += [
        "",
        "## Distribution by domain",
        "| Domain | Files |",
        "|---|---|",
    ]
    for domain in sorted(domain_files.keys()):
        lines.append(f"| {domain} | {domain_files[domain]} |")

    cutoff_display = (now - timedelta(days=since_days)).strftime("%Y-%m-%d")
    lines += [
        "",
        f"## Recent activity (last {since_days} days from log.md, since {cutoff_display})",
        "| Date | Operation | Subject |",
        "|---|---|---|",
    ]
    if recent_log:
        for le in recent_log[-20:]:
            lines.append(f"| {le['date']} | {le['operation']} | {le['subject']} |")
    else:
        lines.append("| — | — | No log entries in this window |")

    lines += [
        "",
        "## Staleness",
        f"- Shelf-index last rebuilt: {last_rebuilt or 'unknown'}",
        f"- Status: {staleness_status}",
    ]

    return "\n".join(lines) + "\n"


def main(args: list[str] | None = None) -> int:
    """CLI entry point. Returns exit code."""
    parser = argparse.ArgumentParser(description="Knowledge base statistics dashboard.")
    parser.add_argument("--library-path", default="library", help="Path to library directory")
    parser.add_argument(
        "--shelf-index-path", default=None, help="Override shelf-index path"
    )
    parser.add_argument("--log-path", default=None, help="Override log.md path")
    parser.add_argument("--output", default=None, help="Write to file instead of stdout")
    parser.add_argument(
        "--since", default=None, metavar="ISO-DATE",
        help="Override recent-activity window start (e.g. 2026-04-01)"
    )
    parsed = parser.parse_args(args)

    library_path = Path(parsed.library_path)
    if not library_path.is_dir():
        print(f"Error: library path '{library_path}' does not exist.", file=sys.stderr)
        return 1

    shelf_index_path = (
        Path(parsed.shelf_index_path) if parsed.shelf_index_path
        else library_path / "_shelf-index.md"
    )
    log_path = (
        Path(parsed.log_path) if parsed.log_path
        else library_path / "log.md"
    )

    since_days = 30
    if parsed.since:
        try:
            since_dt = datetime.strptime(parsed.since, "%Y-%m-%d")
            delta = datetime.now() - since_dt
            since_days = max(1, delta.days)
        except ValueError:
            print(f"Error: --since '{parsed.since}' is not a valid ISO date (YYYY-MM-DD).", file=sys.stderr)
            return 1

    output = generate_stats(library_path, shelf_index_path, log_path, since_days=since_days)

    if parsed.output:
        Path(parsed.output).write_text(output, encoding="utf-8")
        print(f"Stats written to {parsed.output}")
    else:
        print(output, end="")

    return 0


if __name__ == "__main__":
    sys.exit(main())
```

```python
# plugins/sdlc-knowledge-base/scripts/kb_prepare_batch.py
"""Stage source files into library/raw/ with optional format conversion.

Supports .md (pass-through), .pdf/.docx/.pptx/.xlsx/.html/.csv (markitdown),
and .tex/.epub/.rst/.org (pandoc). Unknown extensions are reported as errors.
"""
from __future__ import annotations

import shutil
import subprocess
import sys
from dataclasses import dataclass, field
from datetime import datetime, timezone
from pathlib import Path


_MARKITDOWN_EXTS = frozenset({".pdf", ".docx", ".pptx", ".xlsx", ".html", ".csv"})
_PANDOC_EXTS = frozenset({".tex", ".epub", ".rst", ".org"})


class ConversionError(Exception):
    pass


@dataclass
class BatchResult:
    staged: int = 0
    skipped: int = 0
    failed: int = 0
    errors: list[str] = field(default_factory=list)


def detect_converter(path: Path) -> str | None:
    """Return 'passthrough', 'markitdown', 'pandoc', or None for unknown."""
    ext = path.suffix.lower()
    if ext == ".md":
        return "passthrough"
    if ext in _MARKITDOWN_EXTS:
        return "markitdown"
    if ext in _PANDOC_EXTS:
        return "pandoc"
    return None


def write_provenance_frontmatter(source: str, converter: str, body: str) -> str:
    """Prepend provenance YAML frontmatter to converted body."""
    now = datetime.now(timezone.utc).isoformat().replace("+00:00", "Z")
    frontmatter = (
        "---\n"
        f"source: {source}\n"
        f"converted_by: {converter}\n"
        f"converted_at: {now}\n"
        "status: raw\n"
        "---\n\n"
    )
    return frontmatter + body


def _run_markitdown(source: Path) -> str:
    """Convert source via markitdown CLI; return markdown text."""
    result = subprocess.run(
        ["markitdown", str(source)],
        capture_output=True,
        text=True,
    )
    if result.returncode != 0:
        raise ConversionError(f"markitdown failed: {result.stderr.strip()}")
    return result.stdout


def _run_pandoc(source: Path) -> str:
    """Convert source via pandoc CLI to markdown; return markdown text."""
    result = subprocess.run(
        ["pandoc", str(source), "-t", "markdown"],
        capture_output=True,
        text=True,
    )
    if result.returncode != 0:
        raise ConversionError(f"pandoc failed: {result.stderr.strip()}")
    return result.stdout


def prepare_batch(
    sources: list[Path],
    target_dir: Path,
    mode: str = "copy",
    converter_override: str | None = None,
    overwrite: bool = False,
) -> BatchResult:
    """Stage source files into target_dir (library/raw/).

    mode: 'copy' (default) or 'move'.
    converter_override: 'markitdown' or 'pandoc' to force for all files.
    overwrite: allow re-staging files that already exist in target_dir.
    """
    result = BatchResult()
    target_dir.mkdir(parents=True, exist_ok=True)

    for source in sources:
        converter = converter_override or detect_converter(source)
        if converter is None:
            result.failed += 1
            result.errors.append(
                f"{source.name}: unsupported extension '{source.suffix}' — "
                "use .md (pass-through), .pdf/.docx/.pptx/.xlsx/.html/.csv (markitdown), "
                "or .tex/.epub/.rst/.org (pandoc)"
            )
            continue

        dest_name = source.with_suffix(".md").name
        dest = target_dir / dest_name

        if dest.exists() and not overwrite:
            result.skipped += 1
            result.errors.append(
                f"{source.name}: target '{dest.name}' already exists — use --overwrite"
            )
            continue

        try:
            if converter == "passthrough":
                body = source.read_text(encoding="utf-8")
                content = write_provenance_frontmatter(str(source), "passthrough", body)
                dest.write_text(content, encoding="utf-8")
                if mode == "move":
                    source.unlink()
            elif converter == "markitdown":
                body = _run_markitdown(source)
                content = write_provenance_frontmatter(str(source), "markitdown", body)
                dest.write_text(content, encoding="utf-8")
                if mode == "move":
                    source.unlink()
            elif converter == "pandoc":
                body = _run_pandoc(source)
                content = write_provenance_frontmatter(str(source), "pandoc", body)
                dest.write_text(content, encoding="utf-8")
                if mode == "move":
                    source.unlink()
            result.staged += 1
        except (ConversionError, OSError, UnicodeDecodeError) as exc:
            result.failed += 1
            result.errors.append(f"{source.name}: {exc}")

    return result


def main(args: list[str] | None = None) -> int:
    """CLI entry point."""
    import argparse

    parser = argparse.ArgumentParser(
        description="Stage source files into library/raw/ with optional format conversion."
    )
    parser.add_argument("sources", nargs="*", help="Source file paths")
    parser.add_argument("--from", dest="manifest", default=None, help="Manifest file (one path per line)")
    parser.add_argument("--target-dir", default="library/raw", help="Target directory (default: library/raw)")
    parser.add_argument("--copy", dest="mode", action="store_const", const="copy", default="copy")
    parser.add_argument("--move", dest="mode", action="store_const", const="move")
    parser.add_argument("--overwrite", action="store_true")
    parser.add_argument("--force-markitdown", dest="converter_override", action="store_const", const="markitdown")
    parser.add_argument("--force-pandoc", dest="converter_override", action="store_const", const="pandoc")
    parsed = parser.parse_args(args)

    source_paths: list[Path] = [Path(s) for s in parsed.sources]
    if parsed.manifest:
        manifest_path = Path(parsed.manifest)
        if not manifest_path.exists():
            print(f"Error: manifest file '{manifest_path}' not found.", file=sys.stderr)
            return 1
        for line in manifest_path.read_text(encoding="utf-8").splitlines():
            line = line.strip()
            if line and not line.startswith("#"):
                source_paths.append(Path(line))

    if not source_paths:
        print("Error: no source files specified. Provide file paths or --from <manifest>.", file=sys.stderr)
        return 1

    target_dir = Path(parsed.target_dir)
    result = prepare_batch(
        source_paths,
        target_dir,
        mode=parsed.mode,
        converter_override=parsed.converter_override,
        overwrite=parsed.overwrite,
    )

    print(f"\nBatch preparation complete:")
    print(f"  Staged:  {result.staged}")
    print(f"  Skipped: {result.skipped}")
    print(f"  Failed:  {result.failed}")
    if result.errors:
        print(f"\nIssues:")
        for err in result.errors:
            print(f"  - {err}")

    return 0 if result.failed == 0 else 1


if __name__ == "__main__":
    sys.exit(main())
```

| Argument | Description |
|---|---|

| `.tex`, `.epub`, `.rst`, `.org` | `pandoc` |
| Anything else | Error per file; rest of batch continues |

```python
# plugins/sdlc-knowledge-base/scripts/kb_ingest_batch.py
"""Manifest management for kb-ingest-batch batch ingestion workflow.

Handles .batch-progress.json read/write (atomic), file discovery,
and dispatch-prompt formatting for agent-knowledge-updater in BATCH_MODE.
"""
from __future__ import annotations

import json
import re
import sys
from datetime import datetime, timezone
from pathlib import Path

_FRONTMATTER_RE = re.compile(r"^---\s*\n(.*?)\n---\s*\n", re.DOTALL)
_STATUS_RE = re.compile(r"^\s*status:\s*(\S+)", re.MULTILINE)


def _now() -> str:
    return datetime.now(timezone.utc).isoformat().replace("+00:00", "Z")


def _read_status(path: Path) -> str:
    """Return the `status:` value from frontmatter, or empty string if absent."""
    try:
        text = path.read_text(encoding="utf-8")
        m = _FRONTMATTER_RE.match(text)
        if not m:
            return ""
        status_m = _STATUS_RE.search(m.group(1))
        return status_m.group(1).strip() if status_m else ""
    except (OSError, UnicodeDecodeError):
        return ""


def discover_raw_files(raw_dir: Path) -> list[Path]:
    """Return .md files in raw_dir that have status: raw frontmatter.

    Excludes .batch-progress.json and non-.md files.
    """
    if not raw_dir.is_dir():
        return []
    files = []
    for p in sorted(raw_dir.glob("*.md")):
        if _read_status(p) == "raw":
            files.append(p)
    return files


def load_manifest(manifest_path: Path) -> dict | None:
    """Load manifest from JSON file; return None if file does not exist."""
    if not manifest_path.exists():
        return None
    try:
        return json.loads(manifest_path.read_text(encoding="utf-8"))
    except (json.JSONDecodeError, OSError):
        return None


def save_manifest(manifest_path: Path, manifest: dict) -> None:
    """Write manifest atomically via temp file + rename."""
    tmp = manifest_path.with_suffix(".json.tmp")
    tmp.write_text(json.dumps(manifest, indent=2), encoding="utf-8")
    tmp.rename(manifest_path)


def build_manifest(
    source_paths: list[Path], existing: dict | None = None
) -> dict:
    """Build or update a batch progress manifest.

    source_paths: files to process (status:raw filtered from caller).
    existing: prior manifest to merge with (completed entries preserved).
    """
    already_done: set[str] = set()
    completed: list[dict] = []
    failed: list[dict] = []

    if existing:
        completed = existing.get("completed", [])
        failed = existing.get("failed", [])
        already_done = {c["path"] for c in completed}

    # Filter to status:raw files not already completed
    pending: list[str] = []
    for p in source_paths:
        str_path = str(p)
        if str_path in already_done:
            continue
        status = _read_status(p)
        if status == "raw":
            pending.append(str_path)

    # Preserve pending items from existing manifest that aren't in source_paths
    if existing:
        existing_pending = existing.get("pending", [])
        source_set = {str(p) for p in source_paths}
        for ep in existing_pending:
            if ep not in source_set and ep not in already_done and ep not in pending:
                pending.append(ep)

    total = len(already_done) + len(failed) + len(pending)
    return {
        "started_at": existing["started_at"] if existing else _now(),
        "total": total,
        "completed": completed,
        "failed": failed,
        "pending": pending,
    }


def mark_completed(manifest: dict, path: str, timestamp: str) -> dict:
    """Move path from pending to completed."""
    manifest = dict(manifest)
    manifest["pending"] = [p for p in manifest["pending"] if p != path]
    manifest["completed"] = list(manifest.get("completed", [])) + [
        {"path": path, "completed_at": timestamp, "status": "ingested"}
    ]
    return manifest


def mark_failed(manifest: dict, path: str, error: str, timestamp: str) -> dict:
    """Move path from pending to failed."""
    manifest = dict(manifest)
    manifest["pending"] = [p for p in manifest["pending"] if p != path]
    manifest["failed"] = list(manifest.get("failed", [])) + [
        {"path": path, "error": error, "attempted_at": timestamp}
    ]
    return manifest


def retry_failed(manifest: dict) -> dict:
    """Move all failed entries back to pending."""
    manifest = dict(manifest)
    for f in manifest.get("failed", []):
        manifest["pending"] = list(manifest.get("pending", [])) + [f["path"]]
    manifest["failed"] = []
    return manifest


def format_batch_dispatch_prompt(
    source_path: Path, library_path: Path, shelf_index_path: Path
) -> str:
    """Format the dispatch prompt for agent-knowledge-updater in BATCH_MODE: create-only."""
    return (
        f"BATCH_MODE: create-only\n\n"
        f"Integrate the following source into the knowledge base. "
        f"You are operating in batch mode: (1) do NOT modify existing library files — "
        f"if the source would update an existing file, record status: conflict-existing-file "
        f"and stop; (2) do NOT run kb-rebuild-indexes; (3) do NOT append to log.md — "
        f"the batch driver will do this once at the end.\n\n"
        f"Source: {source_path}\n"
        f"Library: {library_path}\n"
        f"Shelf-index: {shelf_index_path}\n"
    )


def main(args: list[str] | None = None) -> int:
    """CLI entry point for manifest inspection."""
    import argparse

    parser = argparse.ArgumentParser(description="Inspect kb-ingest-batch manifest.")
    parser.add_argument("manifest_path", help="Path to .batch-progress.json")
    parsed = parser.parse_args(args)

    manifest = load_manifest(Path(parsed.manifest_path))
    if manifest is None:
        print("No manifest found.", file=sys.stderr)
        return 1

    print(f"Started:   {manifest.get('started_at', 'unknown')}")
    print(f"Total:     {manifest.get('total', 0)}")
    print(f"Completed: {len(manifest.get('completed', []))}")
    print(f"Failed:    {len(manifest.get('failed', []))}")
    print(f"Pending:   {len(manifest.get('pending', []))}")
    return 0


if __name__ == "__main__":
    sys.exit(main())
```

# Batch Ingestion

Drive `agent-knowledge-updater` over every staged file in `library/raw/`, with progress tracking, resume support, and a single shelf-index rebuild at the end. This is the second stage of the batch workflow: **prepare** (`kb-prepare-batch`) then **ingest** (this skill).

## Arguments

| Argument | Description |
|---|---|
| (none) | Process all `.md` files in `library/raw/` with `status: raw` frontmatter |
| `<dir>` | Process all `.md` files in this directory |
| `<manifest>` | Process the file paths listed in this manifest |
| `--parallel <N>` | Dispatch up to N agents concurrently (default: 1; max recommended: 5) |
| `--retry-failed` | Re-queue failed entries from a prior run |

## Resume behaviour

Progress is tracked in `library/raw/.batch-progress.json`. On re-invocation:
- Files already in `completed` are skipped
- Files in `failed` are left alone unless `--retry-failed` is passed
- New files found in raw/ (or the manifest) are appended to `pending`
